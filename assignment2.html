# 3D Perspective Plot: Wellbeing vs GDP per Capita and Life Expectancy
x <- HPI_filter$`Life_Expectancy(y)`
y <- HPI_filter$GDP_per_capita
z <- HPI_filter$Wellbeing

# Creating a grid for the perspective plot
x_grid <- seq(min(x, na.rm = TRUE), max(x, na.rm = TRUE), length = 30)
y_grid <- seq(min(y, na.rm = TRUE), max(y, na.rm = TRUE), length = 30)

# fuction for z grid
z_function <- function(x, y) {
  z_val <- (x + y) / 2
  return(z_val)
}

z_grid <- outer(x_grid, y_grid, FUN = z_function)

# 3D perspective plot
par(mfrow = c(1, 1), family = "Palatino",
    cex.main = 1.3, cex.lab = 0.8, font.main = 2, font.lab = 1, font.axis = 1)
persp(x_grid, y_grid, z_grid, theta = 30, phi = 20, col = "lightblue", 
      main = "3D Perspective Plot: Wellbeing vs Life Expectancy and GDP", 
      xlab = "Life Expectancy", ylab = "GDP per Capita", zlab = "Wellbeing")
# Calculating average HPI per continent
HPI_avg_per_continent <- HPI_filter %>%
